const user = JSON.parse(localStorage.getItem("user"));

if (!user || !user.isAdmin) {
  alert("  专砖 砖转 注 .");
  window.location.href = "login.html";
}

fetch("http://localhost:3000/admin/users-data")
  .then(res => res.json())
  .then(data => {
    const tbody = document.querySelector("#userTable tbody");
    tbody.innerHTML = "";

    if (data.length === 0) {
      tbody.innerHTML = "<tr><td colspan='5'> 砖转砖 爪</td></tr>";
      return;
    }

    data.forEach(user => {
      const tr = document.createElement("tr");
      tr.innerHTML = `
        <td>${user.name}</td>
        <td>${user.email}</td>
        <td>${user.totalTransactions}</td>
        <td>${new Date(user.createdAt).toLocaleDateString('he-IL')}</td>
        <td>
          <button class="make-admin-btn" data-email="${user.email}">
            驻 
          </button>
        </td>
      `;
      tbody.appendChild(tr);
    });

    //    驻转专 砖
    document.querySelectorAll(".make-admin-btn").forEach(btn => {
      btn.addEventListener("click", async () => {
        const email = btn.dataset.email;
        const confirmAction = confirm(`转  砖专爪 驻 转 ${email} ?`);
        if (!confirmAction) return;

        try {
          const res = await fetch("http://localhost:3000/admin/make-admin", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ email })
          });

          const result = await res.json();
          if (res.ok) {
            alert(result.message);
            location.reload();
          } else {
            alert("砖: " + result.message);
          }
        } catch (err) {
          console.error("砖 注:", err);
          alert("砖  驻转 砖转砖 ");
        }
      });
    });
  })
  .catch(err => {
    console.error("砖 注转 砖转砖", err);
  });

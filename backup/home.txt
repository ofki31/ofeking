// 📊 דף הבית – סטטיסטיקות חודשיות מהשרת בלבד

const totalEl = document.getElementById("total-expense");
const mostCommonEl = document.getElementById("most-common");
const avgEl = document.getElementById("daily-average");
const savingsEl = document.getElementById("monthly-savings");

const user = JSON.parse(localStorage.getItem("user"));
if (!user) {
  alert("אין משתמש מחובר. נא להתחבר.");
  window.location.href = "login.html";
}

const now = new Date();
const currentMonth = now.getMonth();
const currentYear = now.getFullYear();

let allExpenses = [];

// 🟡 טעינת כל ההוצאות/הכנסות של המשתמש
fetch(`http://localhost:3000/transactions/${user.id}`)
  .then(res => res.json())
  .then(data => {
    allExpenses = data.filter(item => {
      const dateObj = new Date(item.date);
      return (
        dateObj.getMonth() === currentMonth &&
        dateObj.getFullYear() === currentYear
      );
    });

    buildStats(allExpenses);
  })
  .catch(err => {
    console.error("שגיאה בטעינת הנתונים:", err);
  });

// 🧠 חישוב והצגת נתונים
function buildStats(expenses) {
  const total = expenses
    .filter(item => item.type === "expense")
    .reduce((sum, item) => sum + Number(item.amount), 0);

  const totalIncome = expenses
    .filter(item => item.type === "income")
    .reduce((sum, item) => sum + Number(item.amount), 0);

  const monthlySavings = totalIncome - total;

  const categoryCount = {};
  expenses
    .filter(item => item.type === "expense")
    .forEach(item => {
      categoryCount[item.category] = (categoryCount[item.category] || 0) + 1;
    });

  let mostCommonCategory = "אין הוצאות";
  let maxCount = 0;
  for (let cat in categoryCount) {
    if (categoryCount[cat] > maxCount) {
      mostCommonCategory = cat;
      maxCount = categoryCount[cat];
    }
  }

  const uniqueDates = new Set(
    expenses.filter(e => e.type === "expense").map(e => e.date)
  );
  const dailyAverage =
    uniqueDates.size > 0 ? (total / uniqueDates.size).toFixed(2) : 0;

  // 👁️‍ עדכון תצוגה
  if (totalEl) totalEl.textContent = `סה"כ הוצאות החודש: ${total.toLocaleString()} ₪`;
  if (avgEl) avgEl.textContent = `ממוצע יומי: ${dailyAverage} ₪`;
  if (mostCommonEl) mostCommonEl.textContent = `הקטגוריה הנפוצה: ${mostCommonCategory}`;
  if (savingsEl) savingsEl.textContent = `סך החיסכון החודש: ${monthlySavings.toLocaleString()} ₪`;

  if (Object.keys(categoryCount).length > 0) {
    renderCategoryChart(categoryCount);
  }
}

// 🎨 צבעים לקטגוריות מה-localStorage (כמו תמיד)
function getCategoryColor(category) {
  let colors = JSON.parse(localStorage.getItem("categoryColors")) || {};
  return colors[category] || "#ccc";
}

// 🧁 גרף עוגה לפי קטגוריות
function renderCategoryChart(categoryCount) {
  const chartEl = document.getElementById("categoryChart");
  if (!chartEl) return;

  const ctx = chartEl.getContext("2d");

  const data = {
    labels: Object.keys(categoryCount),
    datasets: [
      {
        data: Object.values(categoryCount),
        backgroundColor: Object.keys(categoryCount).map(cat => getCategoryColor(cat)),
        borderColor: "#fff",
        borderWidth: 2
      }
    ]
  };

  new Chart(ctx, {
    type: "pie",
    data: data,
    options: {
      plugins: {
        legend: {
          position: "right",
          labels: {
            font: { size: 14 }
          }
        }
      }
    }
  });
}
